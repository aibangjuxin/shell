counter=0
while true; do
  if [ -f /home/script ]
  then
       echo "Files Present"
       exit 0
  elif [[ "$counter" -gt 5 ]]; then
       echo "Counter: $counter times reached; Exiting loop!"
       exit 1
  else
       counter=$((counter+1))
       echo "Counter: $counter time(s); Sleeping for another half an hour"
       sleep 3
  fi
done
